# Fast Docling Markdown Cleanup Example

This document contains a practical, advanced cleanup function for Markdown generated by Docling.
Goal: remove OCR/page noise while preserving meaningful structure (paragraphs, headings, lists, tables, code blocks).

## What this cleaner does

- Normalizes Unicode and common OCR artifacts (`soft hyphen`, `NBSP`, control chars).
- Detects repeated headers/footers across pages and removes them.
- Filters page-number-like lines and low-value scan artifacts.
- Rebuilds wrapped prose into readable paragraphs.
- Preserves Markdown structure:
  - headings (`#`, setext headings)
  - lists (ordered/unordered)
  - blockquotes
  - tables
  - code fences and indented code
  - links/images/references
- Optionally deduplicates consecutive repeated lines.
- Collapses excessive blank lines.

## Python implementation

```python
import re
import unicodedata
from dataclasses import dataclass
from collections import Counter
from typing import List


@dataclass
class MdCleanupConfig:
    page_boundary_depth: int = 3          # top/bottom non-empty lines checked as header/footer
    repeated_min_pages: int = 3           # line repeated across >= N pages -> candidate
    drop_noise_lines: bool = True
    dedupe_consecutive: bool = True
    keep_form_feed: bool = False          # keep \f page separators in output
    collapse_blank_max: int = 2


def clean_docling_markdown(text: str, cfg: MdCleanupConfig = MdCleanupConfig()) -> str:
    """
    Advanced cleanup of Docling markdown:
    - Unicode/control-char normalization
    - Header/footer detection by repeated boundary lines across pages
    - OCR/page artifact filtering
    - Paragraph reconstruction while preserving markdown blocks
    """
    text = _normalize_text(text)
    pages = _split_pages(text)

    repeated_boundary = _detect_repeated_boundary_lines(
        pages, depth=cfg.page_boundary_depth, min_pages=cfg.repeated_min_pages
    )

    cleaned_pages = []
    for page in pages:
        cleaned_pages.append(
            _remove_page_noise(
                page,
                repeated_boundary=repeated_boundary,
                depth=cfg.page_boundary_depth,
                drop_noise_lines=cfg.drop_noise_lines,
            )
        )

    separator = "\n\f\n" if cfg.keep_form_feed else "\n\n"
    text2 = separator.join("\n".join(p) for p in cleaned_pages)

    lines = text2.splitlines()
    lines = _rebuild_paragraphs_preserving_markdown(lines)

    if cfg.dedupe_consecutive:
        lines = _dedupe_consecutive(lines)

    out = _collapse_blank_lines(lines, cfg.collapse_blank_max)
    return out.strip() + "\n"


def _normalize_text(text: str) -> str:
    text = text.replace("\r\n", "\n").replace("\r", "\n")
    text = unicodedata.normalize("NFKC", text)
    text = text.replace("\u00ad", "")  # soft hyphen
    text = text.replace("\xa0", " ")   # nbsp
    text = "".join(ch for ch in text if ch in "\n\t\f" or ord(ch) >= 32)
    text = re.sub(r"([A-Za-z?-??-???0-9])-\n([A-Za-z?-??-???0-9])", r"\1\2", text)
    return text


def _split_pages(text: str) -> List[List[str]]:
    if "\f" in text:
        return [p.splitlines() for p in text.split("\f")]

    page_break_re = re.compile(
        r"(?im)^\s*(?:---+\s*)?(?:page\s+\d+(?:\s*/\s*\d+)?)\s*(?:---+)?\s*$"
    )
    pages = []
    cur = []
    for ln in text.splitlines():
        if page_break_re.match(ln):
            pages.append(cur)
            cur = []
        else:
            cur.append(ln)
    pages.append(cur)
    return pages


def _detect_repeated_boundary_lines(pages: List[List[str]], depth: int, min_pages: int) -> set:
    counts = Counter()

    for page in pages:
        nonempty_idx = [i for i, ln in enumerate(page) if ln.strip()]
        if not nonempty_idx:
            continue

        top = nonempty_idx[:depth]
        bot = nonempty_idx[-depth:] if len(nonempty_idx) > depth else []

        for idx in top + bot:
            norm = _normalize_repeat_key(page[idx])
            if norm:
                counts[norm] += 1

    return {k for k, v in counts.items() if v >= min_pages}


def _normalize_repeat_key(line: str) -> str:
    s = line.strip().lower()
    if not s:
        return ""
    s = re.sub(r"\d+", "#", s)
    s = re.sub(r"\s+", " ", s)
    s = re.sub(r"[^\w\s#:/().,%\-]", "", s, flags=re.UNICODE)
    return s.strip()


def _remove_page_noise(page: List[str], repeated_boundary: set, depth: int, drop_noise_lines: bool) -> List[str]:
    out = []
    nonempty_idx = [i for i, ln in enumerate(page) if ln.strip()]
    top_set = set(nonempty_idx[:depth])
    bot_set = set(nonempty_idx[-depth:]) if len(nonempty_idx) > depth else set()

    for i, ln in enumerate(page):
        s = ln.strip()
        if not s:
            out.append("")
            continue

        norm = _normalize_repeat_key(ln)

        if (i in top_set or i in bot_set) and norm in repeated_boundary:
            continue
        if _looks_like_page_number(s):
            continue
        if _looks_like_watermark_or_artifact(s):
            continue
        if drop_noise_lines and _is_noise_line(s):
            continue

        out.append(ln)

    return out


def _looks_like_page_number(s: str) -> bool:
    patterns = [
        r"^(?:page|???\.?)\s*\d+\s*(?:/|of)?\s*\d*$",
        r"^\d+\s*/\s*\d+$",
        r"^\[\s*\d+\s*\]$",
        r"^-\s*\d+\s*-$",
        r"^\d+$",
    ]
    ss = s.lower()
    return any(re.fullmatch(p, ss) for p in patterns)


def _looks_like_watermark_or_artifact(s: str) -> bool:
    ss = s.lower()
    bad_tokens = ("confidential", "copyright", "all rights reserved", "scanned by")
    return any(tok in ss for tok in bad_tokens) and len(ss) < 120


def _is_noise_line(s: str) -> bool:
    if _is_structural_line(s):
        return False

    if len(s) >= 8:
        alnum = sum(ch.isalnum() for ch in s)
        ratio = alnum / max(len(s), 1)
        if ratio < 0.30:
            return True

    if re.fullmatch(r"(.)\1{6,}", s):
        return True

    return False


def _rebuild_paragraphs_preserving_markdown(lines: List[str]) -> List[str]:
    out = []
    buf = []
    in_code_fence = False

    def flush_buf():
        nonlocal buf
        if not buf:
            return
        out.extend(_merge_paragraph_buffer(buf))
        buf = []

    n = len(lines)
    for i, ln in enumerate(lines):
        s = ln.strip()
        nxt = lines[i + 1].strip() if i + 1 < n else ""

        if re.match(r"^\s*```", ln):
            flush_buf()
            out.append(ln.rstrip())
            in_code_fence = not in_code_fence
            continue

        if in_code_fence:
            out.append(ln.rstrip())
            continue

        if not s:
            flush_buf()
            out.append("")
            continue

        if _is_structural_line(s) or _is_setext_underline(s):
            flush_buf()
            out.append(ln.rstrip())
            continue

        if _is_setext_underline(nxt):
            flush_buf()
            out.append(ln.rstrip())
            continue

        buf.append(ln.rstrip())

    flush_buf()
    return out


def _merge_paragraph_buffer(buf: List[str]) -> List[str]:
    if not buf:
        return []

    merged = [buf[0].strip()]

    for cur_raw in buf[1:]:
        cur = cur_raw.strip()
        prev = merged[-1]

        if _should_keep_newline_between(prev, cur):
            merged.append(cur)
            continue

        if prev.endswith("-") and _is_word_start(cur):
            merged[-1] = prev[:-1] + cur
        else:
            merged[-1] = prev + " " + cur

    return merged


def _should_keep_newline_between(prev: str, cur: str) -> bool:
    if prev.endswith("  ") or prev.endswith("\\"):
        return True
    if prev.endswith(":") and len(prev) <= 80:
        return True
    if _looks_like_title_line(prev) and len(prev.split()) <= 12:
        return True
    if _is_structural_line(cur):
        return True
    return False


def _looks_like_title_line(s: str) -> bool:
    if re.search(r"[.!?]$", s):
        return False
    words = re.findall(r"[A-Za-z?-??-???0-9]+", s)
    if not words or len(words) > 12:
        return False
    caps = sum(w[:1].isupper() for w in words if w)
    return caps / len(words) >= 0.7


def _is_word_start(s: str) -> bool:
    return bool(re.match(r"^[A-Za-z?-??-???0-9]", s))


def _is_structural_line(s: str) -> bool:
    return any([
        bool(re.match(r"^#{1,6}\s+\S", s)),
        bool(re.match(r"^\s*[-*+]\s+\S", s)),
        bool(re.match(r"^\s*\d{1,3}[.)]\s+\S", s)),
        bool(re.match(r"^\s*>\s+\S", s)),
        bool(re.match(r"^\s*([-*_])\1{2,}\s*$", s)),
        bool(re.match(r"^\s*\|.*\|\s*$", s)),
        bool(re.match(r"^\s*:?-{3,}:?(?:\s*\|\s*:?-{3,}:?)+\s*$", s)),
        bool(re.match(r"^\s{4,}\S", s)),
        bool(re.match(r"^\s*!\[.*\]\(.*\)\s*$", s)),
        bool(re.match(r"^\s*\[.{1,80}\]:\s+\S+", s)),
    ])


def _is_setext_underline(s: str) -> bool:
    return bool(re.match(r"^\s*(=+|-+)\s*$", s))


def _dedupe_consecutive(lines: List[str]) -> List[str]:
    if not lines:
        return lines
    out = [lines[0]]
    for ln in lines[1:]:
        if ln.strip() and out[-1].strip() and ln.strip() == out[-1].strip():
            continue
        out.append(ln)
    return out


def _collapse_blank_lines(lines: List[str], max_blank: int) -> List[str]:
    out = []
    blank_run = 0
    for ln in lines:
        if ln.strip():
            blank_run = 0
            out.append(ln.rstrip())
        else:
            blank_run += 1
            if blank_run <= max_blank:
                out.append("")
    return out
```

## Quick usage

```python
from pathlib import Path

raw_md = Path("docling_output.md").read_text(encoding="utf-8")
clean_md = clean_docling_markdown(raw_md)
Path("docling_output.cleaned.md").write_text(clean_md, encoding="utf-8")
```

## Tuning tips

- If real content is being removed too aggressively:
  - increase `repeated_min_pages` to `4` or `5`
  - set `drop_noise_lines=False`
- If headers/footers still leak:
  - increase `page_boundary_depth` to `4` or `5`
- If page boundaries matter for downstream processing:
  - set `keep_form_feed=True`
